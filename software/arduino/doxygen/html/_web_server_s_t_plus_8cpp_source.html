365 
366  for (int i = 0; i < MAX_SOCK_NUM; i++) {
367  uint8_t s = W5100.readSnSR(i);

369  if((s == 0x17) || (s == 0x1C)) {

372  DB1L((i));
373  close(i);
374  }
